125 // blends the foreground and background colour
126 // the symbol is overwritten by the other tiles symbol, unless the symbol value is NULL ('\0')
127 Tile& blend(const Tile& other)
128 {
129 // tile is empty, nothing should be done
130 if (other.is_empty)
131 {
132 return *this;
133 }
134 else if (is_empty)
135 {
136 *this = other;
137 return *this;
138 }
139
140 if (other.symbol != '\0') symbol = other.symbol;
141 // TODO should background colour be blended here aswell?
142 colour.blend(other.colour);
143 background_colour.blend(other.background_colour);
144
145 return *this;
146 }
